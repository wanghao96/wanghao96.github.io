using namespace std;

class Observer
{ public:
explicit Observer(const string strName)
{
this->m_strName = strName;
}
virtual ~Observer() {}
virtual void UpdateState(const string strNotifyInfo) = 0;
protected:
string m_strName; //观察者姓名
};

//市场部人员，具体观察者
class MarketEmployee :public Observer
{ public:
explicit MarketEmployee(const string strName) :Observer(strName){}
virtual void UpdateState(const string strNotifyInfo) override
{
cout << “市场部” << this->m_strName << “收到通知：” << strNotifyInfo << endl;
if (“研发部加班” == strNotifyInfo)
{
cout << “市场部” << this->m_strName << “太好了，可以正常下班了!” << endl;
}
else if (“全体加班” == strNotifyInfo)
{
cout << “市场部” << this->m_strName << “崩溃了，又要加班！” << endl;
}
}
};

//研发部人员，具体的观察者
class DevelopEmployee :public Observer
{ public:
explicit DevelopEmployee(const string strName) :Observer(strName){}
virtual void UpdateState(const string strNotifyInfo) override
{
cout << “研发部” << this->m_strName << “收到通知：” << strNotifyInfo << endl;
if (“研发部加班” == strNotifyInfo)
{
cout << “研发部” << this->m_strName << “崩溃了，又是咱们部门加班！” << endl;
}
else if (“全体加班” == strNotifyInfo)
{
cout << “研发部” << this->m_strName << “好吧，又加班！不过还好有其他人陪。” << endl;
}
}
};

//抽象通知者，提供基本接口定义
class Subject
{ public:
virtual ~Subject(){}
virtual void Attach(Observer* pObserver) = 0;
virtual void Detach(Observer* pObserver) = 0;
virtual void Notify(const string strNotifyInfo) = 0;
protected:
list<Observer*> m_ObserverList; //观察者列表
};

//具体通知者，总经理助理
class Assistant :public Subject
{ public:
void Attach(Observer pObserver)
{
if (pObserver != nullptr)
{
m_ObserverList.push_back(pObserver);
}
}
void Detach(Observer *pOberser)
{
if (pOberser != nullptr)
{
this->m_ObserverList.remove(pOberser);
}
}
void Notify(const string strNotifyInfo)
{
list<Observer>::iterator iter;
for (iter = m_ObserverList.begin(); iter != m_ObserverList.end(); ++iter)
{
(*iter)->UpdateState(strNotifyInfo);
}
}
};

int main()
{ //创建通知者，即总经理助理
Subject* pAssistant = new Assistant();

//创建观察者
Observer* pXiaoWang = new MarketEmployee("小王");
Observer* pXiaoLi = new DevelopEmployee("小李");
Observer* pXiaoZhang = new DevelopEmployee("小张");

//加入观察队列
pAssistant->Attach(pXiaoWang);
pAssistant->Attach(pXiaoLi);
pAssistant->Attach(pXiaoZhang);

//事件1
pAssistant->Notify("全体加班");
cout << endl;

//事件2
pAssistant->Notify("研发部加班");
cout << endl;

//小张和助理闹矛盾了，助理将小张移出通知序列
pAssistant->Detach(pXiaoZhang);

//新事件
pAssistant->Notify("全体加班");
cout << endl;

delete pXiaoWang;
pXiaoWang = nullptr;

delete pXiaoLi;
pXiaoLi = nullptr;

delete pXiaoZhang;
pXiaoZhang = nullptr;

return 0;

}

//传统策略模式实现
class Hurt
{ public:
virtual void redBuff() = 0;
virtual ~Hurt(){}
};

class AdcHurt :public Hurt
{ public:
virtual void redBuff() override
{
cout << “Adc hurt” << endl;
}
};

class ApcHurt :public Hurt
{ public:
virtual void redBuff() override
{
cout << “Apc hurt” << endl;
}
};

//方法1：传入一个指针参数
class Soldier
{ public:
explicit Soldier(Hurt* hurt) :m_hurt(hurt){}
~Soldier(){}
void beInjured()
{
m_hurt->redBuff();
}
private:
Hurt* m_hurt;
};

//方法2：传入一个参数标签
typedef enum
{ ADC,
APC
}HurtType;

class Master
{ public:
explicit Master(HurtType type)
{
switch (type)
{
case ADC:
m_hurt = new AdcHurt;
break;
case APC:
m_hurt = new ApcHurt;
break;
default:
m_hurt = nullptr;
break;
}
}
~Master()
{
if (m_hurt != nullptr)
{
delete m_hurt;
}
}
void beInjured()
{
if (m_hurt != nullptr)
{
m_hurt->redBuff();
}
else
{
cout << “Not hurt” << endl;
}
}
private:
Hurt* m_hurt;
};

    m_fun(num);
